### 安装

```
npm install express --save
```



### 引入

```js
let express = require("express");
```



### 使用

```js
//搭建服务
let app = express();

//监听端口
app.listen(3000,function () {
    console.log('3000端口监听成功');
});

//开放demo目录下的所有资源,自动将static目录下的index作为根路由
app.use(express.static('static'));

//监听路由
app.get('/user', function(req, res){
    //获取get请求发送的数据
    let data = req.query;
    
    res.send('返回数据给前端')
})

//监听路由
app.post('/login', function(req, res){
    //获取post请求发送的数据
    let data = req.body;
    
    res.send('返回数据给前端')
})
```

### 动态路由

```js
// :id  用id这个变量接收article路由的下一级路由名
app.get("/article/:id", function (req, res) {
    //获取article的下级路由名
    let {id} = req.params;
    /*
    	id就是artucle路由的下一级路由，
    	如果需要三级路由："/article/:id/:xx",
    	如果需要四级路由，就在后面再加一个变量
    */
    console.log(id);

    res.send("访问成功")
});
```

### express的洋葱模型

```js
/*
* 执行结果: 1 3 5 6 4 2
* next执行后,会走到下一个中间件,走到最后没有了会找上一个中间件
* */
app.get("/", function (req, res, next) {
    console.log(1);
    //执行下一个中间件函数
    next();
    console.log(2);
});
app.get("/", function (req, res, next) {
    console.log(3);
    //执行下一个中间件函数
    next();
    console.log(4);
});
app.get("/", function (req, res, next) {
    console.log(5);
    //执行下一个中间件函数
    next();
    console.log(6);
});
```
