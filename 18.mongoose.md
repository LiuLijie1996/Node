### 安装

```
npm install mongoose --save
	mongoose这个模块的作用是在node中操作mongodb数据库
```



### 引入

```js
let mongoose = require("mongoose");
```



### 连接数据库

```js
mongoose.connect(
    'mongodb://localhost:27017/blog',
    {
        useNewUrlParser: true,
        useUnifiedTopology: true
    }
).then(() => {
    console.log('数据库连接成功');
}).catch(err => {
    console.log('数据库连接失败');
});
```

------



### 创建表规则

```js
//先定义blog这个数据库中的user表的规则
let UserSchema = mongoose.Schema({
    name: {
        type: String,//数据类型
        required: true,//表示是否必须写
    },
    age: {
        type: Number,//数据类型
    },
    info:{
        sex:String,
        height:Number
    }
});

//让User这个表应用UserSchema
let User = mongoose.model('user', UserSchema);
```

### 改

```js
//updateOne 修改一条符合条件的数据
//updateMany 修改所有符合条件的数据
User.updateOne(
    {//查找数据
        name:"邹秀英"
    },
    {//修改哪些数据
        age:20
    }
).then((data)=>{
    // 修改成功 { n: 1, nModified: 1, ok: 1 }
    // nModified 修改了多少个
    // ok 成功了多少个
    console.log(data);
}).catch(err=>{
    console.log(err);
});



User.updateOne(
    {//查找数据
        name:"邹秀英"
    },
    {
        //修改info下面的sex数据
        $set:{
            "info.sex":"女"
        },
        
        //在arr的最后一位添加数据
        $push:{arr:'abc'},

        //修改arr的第一位
        $set:{"arr.0":'666'},
        
        //保留倒数6个数据,其他全部剔除
        $push: {arr: {$each:[], $slice:-6}},
        
        //如果没有 abc 就添加进arr中,如果有那就不添加
        $addToSet: {arr: 'abc'},
        
        //如果是1删除arr的最后一项, 如果是-1删除arr的第一项
        $pop:{arr:1},
        
        //删除指定项
        $pull:{arr:'abc'},
        
        //自增1
        $inc:{age:1},

        //移除info.sex属性
        $unset:{"info.sex":0},
    }
).then((data)=>{
    console.log(data);
}).catch(err=>{
    console.log(err);
});
```

### 查

```js
//find 查找所有符合条件的数据
//findId 通过id查找数据
//findOne 查找一条符合条件的数据
User.find()
    // .countDocuments()//加上这个函数后,只会获取数据的个数,不会获取具体数据
    .sort({age:-1})//将查找到的数据按照age降序排列
    .then((data) => {
        console.log(data);
    }).catch(err => {
    console.log('查询失败');
});
```

+ 普通查询

  ```js
  User.find(
      {
          age: 42
      },
      (err, data)=>{
          //返回符合字段的所有数据
          console.log(data);
      }
  );
  ```

+ 条件查询

  ```js
  User.find(
      {
          //$gt(大于)  $gte(大于等于)  $lt(小于)  $lte(小于等于)  $ne(不等于)
          age: {$lt:42}
      },
      (err, data)=>{
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  
  
  
  User.find(
      {
          //$or(或)  $nor(取反)
          $or:[{name:"马艳"}, {name:"孙秀兰"}]
      },
      (err, data)=>{
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  
  
  
  User.find(
      {
          //$in(在)  $nin(不在)
          name:{$in:['周芳', '杜明']}
      },
      (err, data)=>{
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  
  
  
  User.find(
      {
          //匹配name字段存在的数据
          name:{$exists:true}
      },
      (err, data)=>{
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  
  
  
  User.find(
      {
          //匹配数组长度是3的数据
          arr:{$size:3}
      },
      (err, data)=>{
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  
  
  
  User.find(
      {
          //数组中是否存在指定的数据
          arr:{$all:['123', '321']}
      },
      (err, data)=>{
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  
  
  
  User.find(
      {
          //使用js语法匹配数据,但是再函数内部取不到外部的变量
          $where:  function(){
              return this.age === 42;
          }
          
          //使用字符串的方式
          $where:`this.article_title.includes('${article_title}')`
      },
      (err, data) => {
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  
  
  
  User.find(
      {
          //使用正则表达式匹配数据
          name: /敏$/
      },
      (err, data) => {
          //返回符合条件的所有数据
          console.log(data);
      }
  );
  ```

### 分页

```js

    {},
    {

    console.log(data);
});
```
